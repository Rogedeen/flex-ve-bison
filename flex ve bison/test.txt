-- Koþullu ifade

    sonuc <- 1

    sonuc <- 0

-- Döngü örneði
dongu x <> 1 iken
    eger x % 2 == 0 ise
        x <- x / 2
    aksi halde
        x <- 3 * x + 1
neki

    sonuc <- a + b
